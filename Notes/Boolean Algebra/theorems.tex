\section*{Boolean Algebra Theorems}

\begin{theorem}
    $X + XY = X$
\end{theorem}

\begin{theorem}
    $X + X'Y = X + Y$
\end{theorem}

\begin{theorem}
    $XY + X'Z + YZ = XY + X'Z$
\end{theorem}

\begin{theorem}
    $X (X + Y) = X$
\end{theorem}

\begin{theorem}
    $X (X' + Y) = XY$
\end{theorem}

\begin{theorem}
    $(X + Y)(X + Y') = X$
\end{theorem}

\begin{theorem}
    $(X + Y)(X' + Z) = XZ + X'Y$
\end{theorem}

\begin{theorem}
    $(X + Y)(X' + Z)(Y + Z) = (X + Y)(X' + Z)$
\end{theorem}

\subsection*{DeMorgan's Theorems}

\begin{enumerate}
    \item Complement entire Expression
    \item Complement each Term
    \item Change operation from AND to OR or OR to AND
\end{enumerate}

\begin{theorem}
    $(XYZ)' = X' + Y' + Z'$ 
\end{theorem}

\begin{theorem}
    $(X + Y + Z)' = X'Y'Z'$
\end{theorem}
